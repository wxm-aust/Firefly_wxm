## 选做部分
* 了解switch-case语法
* 了解向函数传入结构体的方法
* 了解存储类型auto,static,extern,register的区别





```c
#include<stdio.h>  
int func(int s[],int *x){  
    int y;  
    y=0;  
    for(int i=0;i<10;i++){  
        if(s[i]>=90) y++;  
        if(s[i]<60) (*x)++;  
    }  
        return y;  
}  
int main(){  
    int s[10];            //s为学生成绩数组  
    int x,y;            //x为不及格人数，y为优秀人数  
    for(int i=0;i<10;i++){  
        scanf("%d",s+i);            }  
    y=func(s,&x);  
    printf("%d\n%d",x,y);  
    return 0;  
}
```
